cmake_minimum_required(VERSION 3.5)
project(ecat_pkg)

add_compile_options(-Wall -Wextra -pedantic)

set(etherlab_include /opt/etherlab/include)
set(etherlab_rtdm_lib /opt/etherlab/lib/libethercat_rtdm.so.1.1.0)
set(etherlab_lib /opt/etherlab/lib/libethercat.so.1.1.0)

set(xenomai_include /usr/xenomai/include)
set(xenomai_cobalt_include /usr/xenomai/include/cobalt)
set(xenomai_alchemy_include /usr/xenomai/include/alchemy)
set(xenomai_lib_alchemy /usr/xenomai/lib/libalchemy.so.0.0.0)
set(xenomai_lib_copperplate /usr/xenomai/lib/libcopperplate.so.0.0.0)
set(xenomai_lib_cobalt /usr/xenomai/lib/libcobalt.so.2.0.0)
set(xenomai_lib_modechk /usr/xenomai/lib/libmodechk.so.0.0.0)

set(ecat_node_include ~/spinerobot_ws/src/ecat_pkg/include/ecat_pkg/)
set(node_name "ecat_node")


find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(rclcpp_lifecycle REQUIRED)

add_executable(ecat_node src/main.cpp
			 src/ecat_node.cpp
			 src/ecat_slave.cpp)

target_include_directories(ecat_node PUBLIC
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
  $<INSTALL_INTERFACE:include>)

target_link_libraries(ecat_node
	${etherlab_lib}
	${etherlab_rtdm_lib}
	${xenomai_lib_alchemy}
	${xenomai_lib_copperplate}
	${xenomai_lib_cobalt}
	${xenomai_lib_modechk}
)
# Add include directories
include_directories(
  ${etherlab_include}
  ${ecat_node_include}
  ${xenomai_include}
  ${xenomai_cobalt_include}
  ${xenomai_alchemy_include}
)
ament_target_dependencies(${node_name}  rclcpp rclcpp_lifecycle)

install(TARGETS ecat_node
  DESTINATION lib/${PROJECT_NAME})


ament_package()
