23wait_for_result(
24 FutureT & future,
25 WaitTimeT time_to_wait)
26{
27 auto end = std::chrono::steady_clock::now() + time_to_wait;
28 std::chrono::milliseconds wait_period(100);
29 std::future_status status = std::future_status::timeout;
30 do {
31 auto now = std::chrono::steady_clock::now();
32 auto time_left = end - now;
33 if (time_left <= std::chrono::milliseconds(0)) {break;}
34 status = future.wait_for((time_left < wait_period) ? time_left : wait_period);
35 } while (status != std::future_status::ready);
